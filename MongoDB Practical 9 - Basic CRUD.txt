// MongoDB Practical 9 - Basic CRUD Operations
// Aim: To study MongoDB and perform basic Create, Read, Update, and Delete operations.

// 1. Switch to (or create) a database
use Abhi;

// 2. Create a collection
// A collection in MongoDB is like a table in SQL
db.createCollection('Student');

// 3. Insert documents into the collection
// Each document is like a record in JSON format
db.Student.insert({ 'Rno': '1', 'Name': 'Piyush', 'Class': 'TE COMP' });
db.Student.insert({ 'Rno': '2', 'Name': 'Abhi', 'Class': 'TE COMP' });
db.Student.insert({ 'Rno': '3', 'Name': 'Ashley', 'Class': 'TE COMP' });
db.Student.insert({ 'Rno': '4', 'Name': 'Hitesh', 'Class': 'TE COMP' });
db.Student.insert({ 'Rno': '5', 'Name': 'Pratik', 'Class': 'TE COMP' });
db.Student.insert({ 'Rno': '6', 'Name': 'Pratik', 'Class': 'TE COMP' });

// 4. Display all data
// This shows all inserted records
db.Student.find();

// 5. Display data in a readable JSON format
db.Student.find().pretty();

// 6. Update a record
// Here we update the name 'Hitesh' to 'Henry'
db.Student.update({ 'Name': 'Hitesh' }, { $set: { 'Name': 'Henry' } });

// 7. Delete a record
// Removes one record where Name = 'Pratik'
db.Student.remove({ 'Name': 'Pratik' }, 1);

// 8. Check remaining records
db.Student.find().pretty();

// 9. Drop the collection
// Deletes the entire Student collection from the database
db.Student.drop();
